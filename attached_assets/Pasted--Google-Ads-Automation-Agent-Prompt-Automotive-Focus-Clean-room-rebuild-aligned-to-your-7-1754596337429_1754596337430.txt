/**
 * Enhances responses with brand/season context and natural follow-ups
 */
export const enhanceResponse = (
  baseResponse: string,
  context: {
    dealership?: string,
    brand?: string,
    packageType?: string,
    currentMonth?: number,
    userConcerns?: string[]
  }
) => {
  let enhanced = baseResponse;

  if (context.brand) {
    const brandKey = context.brand.toLowerCase();
    if (CONVERSATION_ENHANCERS.brandSpecificInsights[brandKey]) {
      enhanced += `\n\n${CONVERSATION_ENHANCERS.brandSpecificInsights[brandKey]}`;
    }
  }

  const season = getSeason(context.currentMonth ?? new Date().getMonth());
  if (season && Math.random() > 0.7) {
    enhanced += `\n\n${CONVERSATION_ENHANCERS.seasonalContext[season]}`;
  }

  if (Math.random() > 0.6) {
    const questionSet = getRelevantQuestions(context.userConcerns);
    const question = questionSet[Math.floor(Math.random() * questionSet.length)];
    enhanced += `\n\n${question}`;
  }

  return enhanced;
}

function getSeason(month: number): string {
  if (month >= 2 && month <= 4) return 'spring';
  if (month >= 5 && month <= 7) return 'summer';
  if (month >= 8 && month <= 10) return 'fall';
  return 'winter';
}

function getRelevantQuestions(concerns?: string[]): string[] {
  if (!concerns || concerns.length === 0) {
    return CONVERSATION_ENHANCERS.kickoffQuestions;
  }
  const c = concerns.join(' ').toLowerCase();
  if (c.includes('track') || c.includes('convers') || c.includes('ga4') || c.includes('offline')) {
    return CONVERSATION_ENHANCERS.trackingProbes;
  }
  if (c.includes('structure') || c.includes('pmax') || c.includes('search') || c.includes('brand')) {
    return CONVERSATION_ENHANCERS.structureProbes;
  }
  if (c.includes('compet')) {
    return CONVERSATION_ENHANCERS.competitiveIntel;
  }
  return CONVERSATION_ENHANCERS.kickoffQuestions;
}
